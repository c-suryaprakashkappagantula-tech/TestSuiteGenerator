function checkPage(url,domainURL) {
		loopCounter++;
	try{
	
		if(loopCounter == 60){
			OMWindow.location.href = url;
			return;			// maximum recursion loop run time exceeded;
		}
		if(!OMWindow){
			console.info('OMWindow is not a object');
			return;
		}
		var isOpera = !!window.opera || navigator.userAgent.indexOf(' OPR/') >= 0;
		var isChrome = !!window.chrome && !isOpera;   
		console.info('isChrome'+isChrome);
		if(isChrome){
		
			console.info(OMWindow.document.getElementById('applicationName')); //checks the OMpage loadin is completed
		if (OMWindow.document.getElementById('applicationName')) {
			OMWindow.location.href = url; // if loading completed then the URL is changed to the actuall worklistpage or taskpage URL
				return ;
		}
		else{
		setTimeout(function(){
			checkPage(url,domainURL);
		},1000);  // calls the method recursively for every 1 sec and checks if the OM page is loaded
			return;
	}
		
	}
	else{
	
		console.info(OMWindow.document.readyState); //checks the OMpage loadin is completed
		if (OMWindow.document.readyState == 'complete' || OMWindow.document.readyState === "interactive") {
			OMWindow.location.href = url; // if loading completed then the URL is changed to the actuall worklistpage or taskpage URL
				return ;
		}
		else{
		setTimeout(function(){
			checkPage(url,domainURL);
		},1000);  // calls the method recursively for every 1 sec and checks if the OM page is loaded
			return;
	}
	
	}
	}
	catch(err){
		console.info('error in opening the window');
		console.info(err);
		OMWindow.location.href = domainURL+url; // changes the OM page URL to the actual worklistpage or taskpage URL
		return;
	}
}

function spotToggle(value){
    if(value=="true"){
            try
            {
            var inputObjleft = document.getElementsByTagName('input');
            var inputObjheader = parent.document.getElementsByTagName('input');
            for(var i=0;i<inputObjleft.length;i++)
                {
                    if(inputObjleft[i].id=="toAcces")
                    {
                        inputObjleft[i].value="true";
                        inputObjleft[i].focus();
                        break;

                    }

                }
                for(var i=0;i<inputObjheader.length;i++)
                {
                    if(inputObjheader[i].id=="toAcces")
                    {
                        inputObjheader[i].value="true";
                        inputObjheader[i].focus();
                        break;
                    }
                }
            }
            catch(e)
            {
                alert(e.message);
            }

    } else {
            try
            {
            var inputObjleft = document.getElementsByTagName('input');
            var inputObjheader = parent.document.getElementsByTagName('input');

            for(var i=0;i<inputObjleft.length;i++)

                        inputObjleft[i].value="false";

        }

	function loadrecentlyViewed(){
$('#recentViewMask').show();
jQuery.ajax({

//url: "landingAjax.exc?action=loadRecentlyViewed",
url: "loadRecentlyViewed",
type : 'POST',
  dataType: "json",
    success: function(restext){
		$('#recentViewMask').hide();
var recentlyViewedlistHtml="";
			  					for (var i = 0 ; i < restext.length ; i++) {
								var Obj =  restext[i];								
								if(!Obj.isEntityAvailable)
								{				
								var deleteEntity = rvDeleteEntity.replace('glyphiconImage',rvIconArray[Obj.source]);
								deleteEntity = deleteEntity.replace('entityNewVal',Obj.entityNewVal);
								recentlyViewedlistHtml+= deleteEntity;
								}
								else
								{
			var viewedEntity = rvEntity.replace('glyphiconImage',rvIconArray[Obj.source]);
								viewedEntity = viewedEntity.replace('entityId',Obj.entityId);
								viewedEntity = viewedEntity.replace('entityNewVal',Obj.entityNewVal);
								viewedEntity = viewedEntity.replace('source',Obj.source);
								viewedEntity = viewedEntity.replace(/entityNewValue/g,Obj.entityNewVal);
								recentlyViewedlistHtml+= viewedEntity;
								}
								
								}
							
							$('#recentlyviewed').html(recentlyViewedlistHtml);
							$(".scrollbar").perfectScrollbar();
							
if(restext.length==0)
{
recentlyViewedlistHtml+='<li class = "list-group-item valuenull">NO RECENTLY VIEWED ITEMS</li>'				
$('#recentlyviewed').html(recentlyViewedlistHtml);
}
}


});

}

function createCommonForm(action){
	var form = document.createElement("form");
		form.setAttribute("method", "post");
		form.setAttribute("action", action);
		form.setAttribute("target", "RightFrame");
		form.setAttribute("name", "commonActivityForm");
		form.setAttribute("id", "commonActivityForm");
return form;	
}
	
function createElementForFormSubmit(elementName,defaultValue,formDocument){
	var input = document.createElement('input');
		input.type = 'hidden';
		input.name = elementName;
		input.value = defaultValue;
		formDocument.appendChild(input);
}
	
function commonGotoEntityPage(entityId,entityName,source)
{
	$(".toolbarOverlay").click();
	homePageShowMask();

jQuery.ajax({

//url: "landingAjax.exc?action=loadEntityData",
url: "loadEntityData",
type : 'POST',
  dataType: "json",
  data:{source:source,entityId:entityId},
    success: function(restext){
		var url;
		var commonForm;
		var publishFlag = restext.publishFlag;
	if(source=='REPORT')
	{
		var publishFolder="REPORTS"
		if(publishFlag=="Y")
			publishFolder="PUBLISHED REPORTS";

		// url = 'MyReports.exc?action=loadMyReport&queryid='+entityId;
		url = '/centuryreports/loadMyReport';
		commonForm = createCommonForm(url);
		createElementForFormSubmit('queryid',entityId,commonForm);
		var nodeStr = entityId+'##DELIM##'+entityName+'##DELIM##'+restext.type;
		createElementForFormSubmit('nodeString',nodeStr,commonForm);
		createElementForFormSubmit('createdQueryId',entityId,commonForm);
		createElementForFormSubmit('hidUpQueryId',entityId,commonForm);
		createElementForFormSubmit('hidUpQueryName',entityName,commonForm);
		createElementForFormSubmit('actionType',"edit",commonForm);
		createElementForFormSubmit('viewCreated',"viewCreated",commonForm);
		createElementForFormSubmit('sysDirName',publishFolder,commonForm);
		commonChangeToEntityMenu("menuList1");
	}else if(source=='DASHBOARD')
	{	

		//url='dynamicController.exc?action=loadViewDashboardPage&dashboardListingPageFlag=false&currentDashBoardId='+
		//entityId+'&currentPage=Dashboard&mainTabName=Dashboard&backButtonFrom=dashboardListingPageFlag';
		url='/centuryreports/loadViewDashboardPage';
		commonForm = createCommonForm(url);
		createElementForFormSubmit('dashboardListingPageFlag',false,commonForm);
		createElementForFormSubmit('currentDashBoardId',entityId,commonForm);
		createElementForFormSubmit('currentPage','Dashboard',commonForm);
		createElementForFormSubmit('mainTabName', 'Dashboard',commonForm);
		createElementForFormSubmit('backButtonFrom', 'dashboardListingPageFlag',commonForm);
	
		createElementForFormSubmit('dashboardName',entityName,commonForm);
		if(publishFlag=="Y"){
			var currDashTileData={"currentTileDataArray":[{"tileId":"14","tileType":"DASHBOARD","tileLevel":"2"}],"currentTilePath":""}
			var currPubDashTileData={"currentTileDataArrayPD":[{"tileId":"17","tileType":"PUBLISH_DASHBOARD","tileLevel":"2"}],"currentTilePath":""}
			currPubDashTileData.currentTilePath =entityName+"_"+restext.userName;
			createElementForFormSubmit('currDashTileData',JSON.stringify(currDashTileData),commonForm);
			createElementForFormSubmit('currPubDashTileData',JSON.stringify(currPubDashTileData),commonForm);
			createElementForFormSubmit('listingPublishFlag',true,commonForm);
		}
		else
			createElementForFormSubmit('listingPublishFlag',false,commonForm);
		
		commonChangeToEntityMenu("menuList2");
	}else if(source=='LANDING'){
		var appLicationName='Eureka';
		
		if(restext.type=="LANDING_DASHBOARD")
			appLicationName='CPM';
		//url='dynamicLandingController.exc?action=loadViewLandingPage&landingListingPageFlag=false&currentDashBoardId='
		//+entityId+'&currentPage=Landing Page&mainTabName=Landing Page&backButtonFrom=landingListingPageFlag&applicationName='+appLicationName;
		url='/centuryreports/loadViewLandingPage';
		commonForm = createCommonForm(url);
		createElementForFormSubmit('landingListingPageFlag',false,commonForm);
		createElementForFormSubmit('currentDashBoardId',entityId,commonForm);
		createElementForFormSubmit('currentPage','Landing Page',commonForm);
		createElementForFormSubmit('mainTabName','Landing Page',commonForm);
		createElementForFormSubmit('backButtonFrom','landingListingPageFlag',commonForm);
		createElementForFormSubmit('applicationName',appLicationName,commonForm);
				
		createElementForFormSubmit('listingPublishFlag',publishFlag,commonForm);
		createElementForFormSubmit('dashboardName',entityName,commonForm);
		createElementForFormSubmit('currentTab',"PUBLISH_LANDING",commonForm);
		if(publishFlag=="Y")
		{		
	var currDashTileData = {"currentTileDataArray":[{"tileId":"16","tileType":"LANDING_DASHBOARD","tileLevel":"2"}],"currentTilePath":""};
	var currPubDashTileData = {"currentTileDataArrayPD":[{"tileId":"18","tileType":"PUBLISH_LANDING","tileLevel":"2"}],"currentTilePath":""};
		currPubDashTileData.currentTilePath=entityName+"_"+restext.userName;
		createElementForFormSubmit('currDashTileData',JSON.stringify(currDashTileData),commonForm);
		createElementForFormSubmit('currPubDashTileData',JSON.stringify(currPubDashTileData),commonForm);
	}
		commonChangeToEntityMenu("menuList7");
	}
	document.body.appendChild(commonForm);
	commonForm.submit();
}
})
}

function sessionActivator()
{		
	if(timeoutSec <= 0) {
		//window.location.href= loggedOutURL;
		sessionDeactivator();
	}
	else {
		Ext.Ajax.request({
			   //method : 'POST',
			   //url : 'HomePage.exc?action=sessionRefreshed&timeout='+sessionMinutes,
			   url : 'sessionRefreshed',
			   method: 'POST',
			   params: {
	             "timeout" : sessionMinutes
	            },
	            scope:this,
				success : function(response){
				//pageconfirmwindow.close();
				pageconfirmwindow.hide();
				clearInterval(th);
				out.text('120');
				window.setTimeout('checkSessionTimeout()', timeoutReminder);	 //start the timer again
		   },
		   failure : function(){
				sessionDeactivator();
		   }
		  });
	}
}
